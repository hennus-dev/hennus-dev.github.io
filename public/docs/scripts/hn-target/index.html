exports['Laz-target']:addPed('model', 'anim',coords)

exports['Laz-target']:addPed('a_f_y_bevhills_04', 'anim', vector4(100.35, -1077.02, 28.21, 239.06)

exports['Laz-target']:addObject('model',coords)

exports['Laz-target']:addObject('model',coords)
